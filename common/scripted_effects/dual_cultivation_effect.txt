dual_cultivation_effect = {
	# Faster base gain from dual cultivation
	if = {
		limit = { is_cultivator = yes }
		add_cultivation_lifestyle_xp = major_lifestyle_xp
		send_interface_toast = {
			title = trait_1_xp_tt
			desc = trait_1_xp_desc
		}
	}

	# Meridian progression further accelerates lifestyle gain
	if = {
		limit = { has_perk = qi_gathering_meridian_1_perk }
		add_cultivation_lifestyle_xp = minor_lifestyle_xp
	}
	if = {
		limit = { has_perk = qi_gathering_meridian_2_perk }
		add_cultivation_lifestyle_xp = minor_lifestyle_xp
	}
	if = {
		limit = {
			OR = {
				has_perk = qi_gathering_meridian_3a_perk
				has_perk = qi_gathering_meridian_3b_perk
			}
		}
		add_cultivation_lifestyle_xp = medium_lifestyle_xp
	}
	if = {
		limit = { has_perk = qi_gathering_meridian_4_perk }
		add_cultivation_lifestyle_xp = medium_lifestyle_xp
	}

	# Partner also receives accelerated gains
	if = {
		limit = { exists = scope:carn_sex_partner_scope }
		scope:carn_sex_partner_scope = {
			if = {
				limit = { is_cultivator = yes }
				add_cultivation_lifestyle_xp = major_lifestyle_xp
			}
			if = {
				limit = { has_perk = qi_gathering_meridian_1_perk }
				add_cultivation_lifestyle_xp = minor_lifestyle_xp
			}
			if = {
				limit = { has_perk = qi_gathering_meridian_2_perk }
				add_cultivation_lifestyle_xp = minor_lifestyle_xp
			}
			if = {
				limit = {
					OR = {
						has_perk = qi_gathering_meridian_3a_perk
						has_perk = qi_gathering_meridian_3b_perk
					}
				}
				add_cultivation_lifestyle_xp = medium_lifestyle_xp
			}
			if = {
				limit = { has_perk = qi_gathering_meridian_4_perk }
				add_cultivation_lifestyle_xp = medium_lifestyle_xp
			}
		}
	}
}
